#pragma warning (disable:4996)
#include <stdio.h>
#define Max 100

int main() {

	int arr[Max], n = 0, sum = 0, large, small, i = 1;

	FILE *fp = fopen("input10.txt", "r");
	while (fscanf(fp, "%d", &arr[n]) != EOF)
		n++;
	fclose(fp);

	large = arr[n];

	for (i = n; i > 0; i--) {
		if (arr[i] > large) {
			large = arr[i];
		}
		if (arr[i - 1] > arr[i]) {
			sum += large - arr[i];
			large = arr[i - 1];
		}
		else if (i == 1) {
			if (arr[0] < arr[1])
				sum += large - arr[0];
		}
	}
	printf("ÀÌÀÍ : %d", sum);

	return 0;
}
